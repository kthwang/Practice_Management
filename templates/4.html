    dice2 = random.randint(1,6)
    result = dice1 + dice2
    count_list.append(result)

for i in range(2,13):
    print(str(i) + ' occurs ' + str((count_list.count(i) / 1000)*100) + ' theory is ' + str(theory[i-2]))
