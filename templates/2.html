#Compute a Grade Point Average

grade_list = []

def change(x):
    return {
        'A+': 4.0,
        'A': 4.0,
        'A-': 3.7,
        'B+': 3.3,
        'B': 3.0,
        'B-': 2.7,
        'C+': 2.3,
        'C': 2.0,
        'C-': 1.7,
        'D+': 1.3,
        'D': 1.0,
        'F': 0.0,
        '': ''
    }[x]

while True:
    grade = str(input())
    grade_list.append(change(grade))
    if grade is '':
        grade_list.pop()
        break;

a = sum(grade_list)
b = len(grade_list)
c = a / b
print('Your Average Grade is ' + str(c))

#Admission Price

price_list = []

while True:
    try:
        age = input()
        if 0 <= int(age) <= 2 :
            price_list.append(0.00)
        elif 3 <= int(age) <= 12 :
            price_list.append(14.00)
        elif 13 <= int(age) <= 64 :
            price_list.append(23.00)
        elif int(age) >= 65 :
            price_list.append(18.00)        
    except:
        if age is '' :
            break;
        
        
a = sum(price_list)
print('Total Price is ' + str(a))

#Greatest Common Divisor
def checkGCD(num1, num2):
    if num1 < num2:
        num3 = num1
    else:
        num3 = num2 
        
    while True:
        if num2 % num3 == 0 and num1 % num3 == 0:
            print(num3)
            break;
        num3 -= 1
